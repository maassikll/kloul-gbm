#include <LiquidCrystal.h>//pour l'afichage sur le lcd
const int rs=12 ,en=11 ,d4=2,d5=3 ,d6=4,d7=5;
LiquidCrystal lcd(rs ,en ,d4 ,d5 ,d6 ,d7);
const int sensorPin= 0, btn=13;
bool etat;
int deg=19;
void setup()
{
  
  lcd.begin(16,2);
  pinMode(btn, INPUT);
}

void loop()
{
  
  lcd.setCursor(0,0);
  lcd.print("T:");
  lcd.print(GetTemp());
   lcd.setCursor(0,1);
  etat=digitalRead(btn);
  lcd.print("alarme: ");
  
  if(etat==1)
  {
    deg--;
  }
  if(deg<16)
  {
    deg=24;
  }
  lcd.print(deg);
  
  
  delay(1000);
  lcd.clear();
  
 

}

float GetTemp()//pour la convertion 
{
int reading = analogRead(sensorPin);
float voltage = reading*5.0;
voltage/=1024.0;
return  (voltage-0.5)*100;


}
